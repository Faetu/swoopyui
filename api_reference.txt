# To push new updates to client.
URL command: "/"
METHOD: POST only.
data: {
    "update_number" : <int>,
    "action_name" : <str>,
    "action_content" : <dict>
}

# To get new updates from client.
URL command: "/client_side_update"
METHOD: POST only.
data: {
    "update_name" : <str>,
    "update_content" : <dict>
}
